\section*{Problem 3}

\begin{enumerate}[a)]
\item Solve the first order differential equation
$$ \frac{dy}{dx} + 3y = 0 $$
\item Solve the second order differential equation ($A$ and $B$ are constants)
$$ \frac{d^2 y}{dx^2} - A^2y = B $$
The boundary condition is $y(\pm\frac{1}{A}) = 0$.
\end{enumerate}
